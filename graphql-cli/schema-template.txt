const { gql } = require('apollo-server-express');

module.exports = gql`
  type ${name} {
    id: ID!
    title: String!
    author: String!
    body: String!
  }

  extend type Query {
    ${name}(id: ID!): ${name}
    ${name}s(title: String): [${name}!]
  }

  extend type Mutation {
    create${name}(
      title: String!
      author: String!
      body: String!
    ): ${name}!
  }
`;
